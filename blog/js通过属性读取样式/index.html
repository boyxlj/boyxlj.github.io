js通过属性读取样式

2023年1月8日 · 阅读需 4 分钟
Miraitowa
越努力，越幸运

1.clientWidth 和 clientHeight​

  • 获取元素内部的宽度和高度(内容区域 + 内边距);
  • 不包括外边距,边框,以及滚动条的宽或高;
  • 只读属性;

2.scrollWidth 和 scrollHeight​

  • 获取元素滚动区域的大小(内容区域 + 内边距);
  • 不包括外边距,边框,以及滚动条的宽或高;
  • 只读属性;

3.offsetWidth 和 offsetHeight​

  • 获取元素可见区域的大小(内容区域 + 内边距 + 边框);
  • 只读属性;

4.offsetParent​

  • 获取当前元素的定位父元素;
  • 定位父元素:距当前元素最近的且开启定位的祖先元素,如果都没有开启则返回body;
  • 只读属性;

5.offsetTop 和 offsetLeft​

  • 获取当前元素相对于定位父元素的偏移量;
  • 定位父元素:距当前元素最近的且开启定位的祖先元素,如果都没有开启则以body为参照物;
  • 只读属性;

6.scrollTop 和 scrollLeft​

  • 获取或设置滚动条的偏移量;

完整演示代码​

<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta http-equiv="X-UA-Compatible" content="IE=edge">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>通过属性读取样式</title>
  <style>

    #box1{
      position: relative;
      border: 2px solid sienna;
    }

    #box2{
      width: 300px;
      height: 300px;
      padding: 50px;
      border: 10px solid goldenrod;
      margin: 50px;
      background-color: skyblue;
      overflow: auto;
    }

    #box3{
      width: 100px;
      height: 500px;
      background: red;
    }
  </style>
</head>
<body>
  <div id="box1">
    <div id="box2">
      <div id="box3"></div>
    </div>
  </div>

  <button id="btn">get scroll</button>

  <script type="text/javascript">
    const box2 = document.getElementById('box2')

    /*  
        clientWidth 和 clientHeight
          1.获取元素内部的宽度和高度(内容区域 + 内边距);
          2.不包括外边距,边框,以及滚动条的宽或高;
          3.只读属性;
    */

    console.log(`clientWidth,clientHeight`,box2.clientWidth,box2.clientHeight)   // 383 400

    /* 
        scrollWidth 和 scrollHeight
          1.获取元素滚动区域的大小(内容区域 + 内边距);
          2.不包括外边距,边框,以及滚动条的宽或高;
          3.只读属性;
    */

    console.log(`scrollWidth,scrollHeight`,box2.scrollWidth,box2.scrollHeight)   // 383 600

    /* 
        offsetWidth 和 offsetHeight
          1.获取元素可见区域的大小(内容区域 + 内边距 + 边框);
          2.只读属性;
    */

    console.log(`offsetWidth,offsetHeight`,box2.offsetWidth,box2.offsetHeight)   // 420 420

    /* 
        offsetParent
          1.获取当前元素的定位父元素;
          2.定位父元素:距当前元素最近的且开启定位的祖先元素,如果都没有开启则返回body
          2.只读属性;
    */

    console.log(`offsetParent`,box2.offsetParent)  //  <div id="box1">...</div>

    /* 
        offsetTop 和 offsetLeft
          1.获取当前元素相对于定位父元素的偏移量;
          2.定位父元素:距当前元素最近的且开启定位的祖先元素,如果都没有开启则以body为参照物
          2.只读属性;
    */

    console.log(`offsetTop,offsetLeft`,box2.offsetTop,box2.offsetLeft)   // 50 50

    /* 
        scrollTop 和 scrollLeft
          1.获取或设置滚动条的偏移量;
    */

    btn.onclick = function(){
      console.log(`scrollTop,scrollLeft`,box2.scrollTop,box2.scrollLeft)  
    }

    box2.scrollTop = 100

  </script>
</body>
</html>
